    First of all you need add the manager as a property in the discord client, there are two ways I use first you can added when the client is ready, whatch the first example:

    const { Client } = require('discord.js')
const { LevelManager } = require('discord-xplus')

const client = new Client({
    intents: [...],
    partials: [...]
})

client.on('ready', (bot) =>{
    // There are more options in the manager but mongoURI is important
    client.levelManager = new LevelManager({
        mongoURI: 'URL of your database',
        ...
    })
})

client.login('Token')

    But I prefer the second one because in Vscode you can have it in the autocompletion:

    const { Client } = require('discord.js')
const { LevelManager } = require('discord-xplus')

// You can have it in other file
class MyClient extends Client{
    constructor(options){
        this.on('ready', async (bot) =>{
            // I put await because it can load events from a folder
            // and need time to load all the cache
            this._levelManager = await new LevelManager({
                mongoURI: options.mongoURI,
                ...
            })
        })
    }

    get levelManager(){
        return this._levelManager
    }
}

const client = new MyClient({
    intents: [...],
    partials: [...],
    mongoURI: 'URL of your database'
})

client.levelManager.on('managerReady', (bot) =>{
    // Code...
})

client.login('Token')
